; Purpose: Oculta el PJ automáticamente, si es revelado.

set %version 1.1
;
set #SYSMSGCOL 89
event sysmessage Auto-Hider %version
wait 40
;
;========================================================
;
; ******************* LOOP PRINCIPAL ********************
;
;========================================================
;
loop:
gosub check_muerto
if ( H in #CHARSTATUS || #TARGCURS = 1 || #LLIFTEDKIND = 1 ) ; Si ya estamos ocultos, o vamos a hacer click en algo, o estamos levantando un objeto, entonces no hacemos nada.
     {
       wait 20
       goto loop
     }
     else
        {
          if ( H Notin #CHARSTATUS && #TARGCURS = 0 && #LLIFTEDKIND = 0 ) ; Si no estamos ocultos, y no vamos a hacer clicks, ni a levantar un objeto, entonces nos ocultamos...
               {
                 event macro 13 21 ; Use SKILL Hiding
                 gosub waitForSysVar CHARSTATUS = H 50
               }
       }
goto loop
;
;========================================================
;
; ***************** FIN LOOP PRINCIPAL *****************
;
;========================================================
;

sub check_muerto ; <==============================>
_checkDead:
if #CHARGHOST = Yes
   {
     wait 1
     goto _checkDead
   }
return

sub waitForSysVar ; <==============================>
if %0 < 4 || %4 = N/A
set %4 %_defaultWaitForTimeout
if %0 < 5
set %5 return
set %4 #sCnt2 + %4
_waitForSysVar:
set % . %5 # . %1 %2 %3
if ! % . %5 && #sCnt2 < %4
{
Wait 1
goto _waitForSysVar
}
return
